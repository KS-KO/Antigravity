cmake_minimum_required(VERSION 3.16)

project(MariaDBPartitionManager VERSION 0.1 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTORCC ON)
set(CMAKE_AUTOUIC ON)

find_package(Qt6 REQUIRED COMPONENTS Widgets Sql)

add_executable(MariaDBPartitionManager
    main.cpp
    MainWindow.cpp
    MainWindow.h
    DatabaseManager.cpp
    DatabaseManager.h
)

target_link_libraries(MariaDBPartitionManager PRIVATE Qt6::Widgets Qt6::Sql)
